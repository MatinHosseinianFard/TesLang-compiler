def int find(vector<int> A, int n) {
    var int k;
    var float j;
    for (i = 0 to length(A)) {
        if (n == k) {
            return j;
        }
        j = j + 1;
    }
    return -1;
}
def int main() {
    var int A;
    var int a;
    
    A = list(3);
    
    A[0] = 1;
    A[1] = 2;
    A[2] = 3;
    
    print(find(A, a));
    print(find(A));
    print(find(a, A));
    return A;
}
